068    @Inject

071                         DeployRegistry registry) {
072        super(settings);

074        this.environment = environment;
075        this.nodeService = nodeService;
076        this.deployClassLoader = new URIClassLoader(settings.getClass().getClassLoader());

080    @Override
081    protected void doStart() throws ElasticsearchException {
082        try {
083            initializeInstalledPlugins();
084        } catch (IOException e) {
085            throw new ElasticsearchException(e.getMessage(), e);
086        }
087        nodeService.putAttribute("plugins",
088                Strings.collectionToCommaDelimitedString(registry.getPlugins().keySet()));
089        logger.info("started");
090    }

093    protected void doStop() throws ElasticsearchException {
094        nodeService.removeAttribute("plugins");

105    public void initializeInstalledPlugins() throws IOException {
106        File dir = new File(environment.pluginsFile(), DeployPlugin.NAME + File.separator + "plugins");
107        if (dir.isDirectory()) {
108            logger.info("found plugin dir {}", dir.getAbsolutePath());
109            File[] plugins = dir.listFiles();
110            if (plugins != null) {
111                for (File plugin : plugins) {
112                    logger.info("found plugin {}", plugin.getAbsolutePath());
113                    if (plugin.isDirectory()) {
114                        add(plugin.getName(), plugin);
115                    }
116                }
117            }
118        }
119    }

121    /**
122     * Add plugin to registry
123     *
124     * @param name the name to register the plugin under
125     * @param path zip file or jar file
126     * @throws java.io.IOException if method fails
127     */
128    public void add(String name, File path) throws IOException {
129        // create new class loader for each plugin
130        URIClassLoader classLoader = new URIClassLoader(deployClassLoader);
131        // try to unpack zip
132        path = tryUnpackArchive(path);
133        // find all jars in archive
134        Set<URI> jars = findJars(path);
135        Map<URI, String> pluginClassNames = newHashMap();
136        // build class loader, find jar with the es-plugin.properties in it
137        for (URI jar : jars) {
138            classLoader.addURI(jar);
139            URIClassLoader tmp = new URIClassLoader(deployClassLoader);
140            tmp.addURI(jar);
141            Enumeration<URL> urls = tmp.findResources("es-plugin.properties");
142            while (urls.hasMoreElements()) {
143                URL url = urls.nextElement();
144                if (!pluginClassNames.containsKey(jar)) {
145                    Properties props = new Properties();
146                    InputStream in = url.openStream();
147                    props.load(in);
148                    in.close();
149                    pluginClassNames.put(jar, props.getProperty("plugin"));
150                }
151            }
152        }
153        for (URI uri : classLoader.getURIs()) {
154            logger.info("class path member {}", uri);
155        }
156        // instantiate all plugins in this path, add them to registry
157        for (Map.Entry<URI, String> entries : pluginClassNames.entrySet()) {
158            // check for existing plugin
159            if (registry.getPlugin(name) != null) {
160                logger.warn("old plugin {} exists", name);
161                Plugin oldPlugin = registry.getPlugin(name);
162                stopServices(registry.getInjector(name), oldPlugin);
163                logger.info("services stopped for plugin {}", name);
164                registry.removePlugin(name);
165            }
166            logger.info("instantiating plugin {}", name);
167            Plugin plugin = instantiatePluginClass(entries.getValue(), classLoader);
168            Injector injector = processModules(this.injector, plugin);
169            logger.info("modules added for plugin {}", name);
170            startServices(injector, plugin, settings, classLoader, entries.getKey().toURL());
171            logger.info("services started for plugin {}", name);
172            registry.addPlugin(name, plugin, injector);
173        }
174        logger.info("registry after adding = {}", registry);
175        nodeService.putAttribute("plugins",
176                Strings.collectionToCommaDelimitedString(registry.getPlugins().keySet()));
177    }

179    @SuppressWarnings("unchecked")
180    private Plugin instantiatePluginClass(String className, ClassLoader classLoader) {
181        try {
182            Class<? extends Plugin> cl = (Class<? extends Plugin>) classLoader.loadClass(className);
183            try {
184                return cl.getConstructor(Settings.class).newInstance(settings);
185            } catch (NoSuchMethodException e) {
186                try {
187                    return cl.getConstructor().newInstance();
188                } catch (NoSuchMethodException e1) {
189                    throw new NoSuchMethodException("No constructor for [" + cl + "]. A plugin class must " +
190                            "have either an empty default constructor or a single argument constructor accepting a " +
191                            "Settings instance");
192                }
193            }
194        } catch (Throwable e) {
195            throw new ElasticsearchException("Failed to load plugin class [" + className + "]", e);
196        }
197    }

199    private File tryUnpackArchive(File file) throws IOException {
200        if (!file.exists()) {
201            throw new IOException("file does not exist: " + file.getAbsolutePath());
202        }
203        if (!file.canRead()) {
204            throw new IOException("can not read: " + file.getAbsolutePath());
205        }
206        if (!file.getName().toLowerCase().endsWith(".zip")) {
207            return file;
208        }
209        ZipFile zipFile = new ZipFile(file);
210        try {
211            //we check whether we need to remove the top-level folder while extracting
212            //sometimes (e.g. github) the downloaded archive contains a top-level folder which needs to be removed
213            boolean removeTopLevelDir = topLevelDirInExcess(zipFile);
214            Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
215            while (zipEntries.hasMoreElements()) {
216                ZipEntry zipEntry = zipEntries.nextElement();
217                if (zipEntry.isDirectory()) {
218                    continue;
219                }
220                String zipEntryName = zipEntry.getName().replace('\\', '/');
221                if (removeTopLevelDir) {
222                    zipEntryName = zipEntryName.substring(zipEntryName.indexOf('/'));
223                }
224                File target = new File(file.getParent(), zipEntryName);
225                Streams.copy(zipFile.getInputStream(zipEntry), new FileOutputStream(target));
226            }
227            return file.getParentFile();
228        } catch (Exception e) {
229            logger.error("failed to extract " + file.getAbsolutePath(), e);
230        } finally {
231            try {
232                zipFile.close();
233            } catch (IOException e) {
234                // ignore
235            }
236            // remove zip file
237            if (!file.delete()) {
238                logger.warn("could not remove zip file");
239            }
240        }
241        return file.getParentFile();
242    }
243
244    private boolean topLevelDirInExcess(ZipFile zipFile) {
245        // we don't rely on ZipEntry#isDirectory because it might be that there is no explicit dir
246        // but the files path do contain dirs, thus they are going to be extracted on sub-folders anyway
247        Enumeration<? extends ZipEntry> zipEntries = zipFile.entries();
248        Set<String> topLevelDirNames = newHashSet();
249        while (zipEntries.hasMoreElements()) {
250            ZipEntry zipEntry = zipEntries.nextElement();
251            String zipEntryName = zipEntry.getName().replace('\\', '/');
252            int slash = zipEntryName.indexOf('/');
253            // if there isn't a slash in the entry name it means that we have a file in the top-level
254            if (slash == -1) {
255                return false;
256            }
257            topLevelDirNames.add(zipEntryName.substring(0, slash));
258            // if we have more than one top-level folder
259            if (topLevelDirNames.size() > 1) {
260                return false;
261            }
262        }
263        return topLevelDirNames.size() == 1;
264    }

266    private Set<URI> findJars(File root) {
267        Set<URI> jars = newHashSet();
268        File[] list = root.listFiles();
269        if (list == null) {
270            return jars;
271        }
272        for (File f : list) {
273            if (f.isDirectory()) {
274                jars.addAll(findJars(f.getAbsoluteFile()));
275            } else {
276                if (f.getName().endsWith(".jar")) {
277                    jars.add(URI.create("file:" + f.getAbsolutePath()));
278                }
279            }
280        }
281        return jars;
282    }

284    private Injector processModules(Injector injector, Plugin plugin) {
285        List<Module> modules = Lists.newArrayList();
286        for (Class<? extends Module> moduleClass : plugin.modules()) {
287            modules.add(createModule(moduleClass, settings));
288        }
289        for (Module module : modules) {
290            plugin.processModule(module);
291            List<OnModuleReference> references = findOnModuleReferences(plugin);
292            if (references != null) {
293                for (OnModuleReference reference : references) {
294                    if (reference.moduleClass.isAssignableFrom(module.getClass())) {
295                        try {
296                            reference.onModuleMethod.invoke(plugin, module);
297                        } catch (Exception e) {
298                            logger.warn("plugin {} failed to invoke custom onModule method", e, plugin.name());
299                        }
300                    }
301                }
302            }
303        }
304        return injector.createChildInjector(modules);
305    }
306
307    @SuppressWarnings("unchecked")
308    private List<OnModuleReference> findOnModuleReferences(Plugin plugin) {
309        List<OnModuleReference> list = Lists.newArrayList();
310        for (Method method : plugin.getClass().getDeclaredMethods()) {
311            if (!method.getName().equals("onModule")) {
312                continue;
313            }
314            if (method.getParameterTypes().length == 0 || method.getParameterTypes().length > 1) {
315                logger.warn("plugin {} implementing onModule with no parameters or more than one parameter", plugin.name());
316                continue;
317            }
318            Class moduleClass = method.getParameterTypes()[0];
319            if (!Module.class.isAssignableFrom(moduleClass)) {
320                logger.warn("plugin {} implementing onModule by the type is not of Module type {}", plugin.name(), moduleClass);
321                continue;
322            }
323            method.setAccessible(true);
324            list.add(new OnModuleReference(moduleClass, method));
325        }
326        return list;
327    }

334    static class OnModuleReference {
335        public final Class<? extends Module> moduleClass;
336        public final Method onModuleMethod;
337
338        OnModuleReference(Class<? extends Module> moduleClass, Method onModuleMethod) {
339            this.moduleClass = moduleClass;
340            this.onModuleMethod = onModuleMethod;
341        }
342    }

344    private void startServices(Injector injector, Plugin plugin, Settings settings, ClassLoader classLoader, URL url) throws IOException {
345        for (Class<? extends LifecycleComponent> service : plugin.services()) {
346            logger.info("found service {} {} to start", service, service.getClass());
347            LifecycleComponent t = injector.getInstance(service);
348            if (t instanceof DeployableComponent) {
349                DeployableComponent component = (DeployableComponent) t;
350                logger.info("before init component {}", component);
351                component.init(settings, classLoader, url);
352                logger.info("after component {}", component);
353            }
354            logger.info("starting service {}", service);
355            t.start();
356            logger.info("service {} started", service);
357        }
358    }
359
360    private void stopServices(Injector injector, Plugin plugin) {
361        for (Class<? extends LifecycleComponent> service : plugin.services()) {
362            logger.info("found service {} {} to stop", service, service.getClass());
363            injector.getInstance(service).stop();
364            logger.info("service {} stopped", service);
365        }
366    }
